ng generate @angular/material:material-nav --name=main-nav

@import '~material-design-icons/iconfont/material-icons.css';
@import "~bootstrap/dist/css/bootstrap-grid.min.css";
html, body{
  height: 100%;
}
button:focus {
  outline: none;
}
